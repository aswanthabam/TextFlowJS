          <script src="https://cdn.jsdelivr.net/gh/aswanthabam/TextFlowJS@main/textWrap.js" type="text/javascript" charset="utf-8">
        </script>
        <!-- Script for the page -->
        <script>
            <!--Initializing new wrapper -->
            obj.Elements[i] = new initWrap();
            <!-- Change text size -->
            obj.Elements[i].wrapTextSize = 20;
            <!-- Change Text Color -->
            obj.Elements[i].wrapTextColor = "#aed7d6";
            <!-- Change cursor blinking frequency -->
            obj.Elements[i].blinkFrequency = 500;
            <!-- Change cursor color -->
            obj.Elements[i].CursorBackgroundColor = "#f6f6f7";
            <!-- Change Font family -->
            obj.Elements[i].wrapFontFamily = "Monospace";
            <!-- Start wrapping you can give text and element as arguments -->
            <!-- Automatically get the element and text from the class -->
            <!-- Syntex "setWrap(text = null,elem = null);" -->
            obj.Elements[i].setWrap();
        </script>
    </body>
